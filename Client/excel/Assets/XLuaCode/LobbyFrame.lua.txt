--打开登陆界面
local frame = nil

--点击返回按钮
local OnClickReturn = function()
	if nil ~= frame then
		CS.GameClient.UIFrameLua.CloseFrame(frame)
		frame = nil
	end
end

function OnOpenFrame(x)
	--添加点击登陆安扭事件
	frame = x
	x = nil
	frame:GetObject('btnReturn').onClick:AddListener(OnClickReturn)
end

function OnCloseFrame(x)
	frame = nil
end
